# 호이스팅

## 함수선언의 호이스팅
```jsx
function a() {
    console.log(b) // undefined일거라고 생각이 들지만, function b()가 호출된다. 그 이유는 호이스팅
    let b = 'bbb';
    console.log(b); // 'bbb'
    function b(){ // 함수부분만 위로 끌어올리고 
        console.log(b) // 여기는 가만히 있는다. 'bbb'가 찍힌다.
    }
}
a();
```
